1function [test_results] = applyKerasTrainedModel(test_data,weight,NodesArray)
 2%--------------------------------------------------------------------------
 3% [Output] = applyKerasTrainedModel(test_data,net)
 4%
 5% applying Keras trained neural network to fit input where:
 6%   -net: Keras Neural Network Object as numpy array (2 layers)
 7%   -Input: Testing Input
 8% 
 9% based on Keras with Tensorflow, python 3.6, all activation function uses
10% sigmoid (copy sigmoid.m together with this file for use elsewhere)
11%--------------------------------------------------------------------------
12% below line for avoiding values < 1 in dll application which cause system
13% access violation exception
14test_data = test_data - 1;
15in_size = length(test_data);
16cursor = 1;
17Layers = length(NodesArray);
18temp = test_data';
19temp = sigmoid(reshape(weight(cursor:cursor + in_size*NodesArray(1)-1),NodesArray(1),in_size)*temp+weight(cursor+in_size*NodesArray(1):cursor+in_size*NodesArray(1)+NodesArray(1)-1)');
20cursor = cursor + in_size*NodesArray(1);
21cursor = cursor + NodesArray(1);
22for i = 2:Layers
23    temp = sigmoid(reshape(weight(cursor:cursor + NodesArray(i-1)*NodesArray(i)-1),NodesArray(i),NodesArray(i-1))*temp+weight(cursor+ NodesArray(i-1)*NodesArray(i):cursor+ NodesArray(i-1)*NodesArray(i)+NodesArray(i)-1)');
24    cursor = cursor + NodesArray(i-1)*NodesArray(i);
25    cursor = cursor + NodesArray(i);
26end
27temp = sigmoid(weight(cursor:cursor + NodesArray(Layers)-1)*temp+weight(end));
28test_results = temp;
29
30end
