1function [test_results] = applyKerasTrainedModel(test_data,weight,NodesArray)
 2%--------------------------------------------------------------------------
 3% [Output] = applyKerasTrainedModel(test_data,net)
 4%
 5% applying Keras trained neural network to fit input where:
 6%   -net: Keras Neural Network Object as numpy array (2 layers)
 7%   -Input: Testing Input
 8% 
 9% based on Keras with Tensorflow, python 3.6, all activation function uses
10% sigmoid (copy sigmoid.m together with this file for use elsewhere)
11%--------------------------------------------------------------------------
12
13in_size = length(test_data);
14cursor = 1;
15Layers = length(NodesArray);
16temp = test_data';
17temp = sigmoid(reshape(weight(cursor:cursor + in_size*NodesArray(1)-1),NodesArray(1),in_size)*temp+weight(cursor+in_size*NodesArray(1):cursor+in_size*NodesArray(1)+NodesArray(1)-1)');
18cursor = cursor + in_size*NodesArray(1);
19cursor = cursor + NodesArray(1);
20for i = 2:Layers
21    temp = sigmoid(reshape(weight(cursor:cursor + NodesArray(i-1)*NodesArray(i)-1),NodesArray(i),NodesArray(i-1))*temp+weight(cursor+ NodesArray(i-1)*NodesArray(i):cursor+ NodesArray(i-1)*NodesArray(i)+NodesArray(i)-1)');
22    cursor = cursor + NodesArray(i-1)*NodesArray(i);
23    cursor = cursor + NodesArray(i);
24end
25temp = sigmoid(weight(cursor:cursor + NodesArray(Layers)-1)*temp+weight(end));
26test_results = temp;
27
28end
